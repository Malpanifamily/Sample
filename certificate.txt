  <script>
    const dateInput = document.getElementById('date');
    if(!dateInput.value){
      const today = new Date();
      const yyyy = today.getFullYear();
      const mm = String(today.getMonth()+1).padStart(2, '0');
      const dd = String(today.getDate()).padStart(2, '0');
      dateInput.value = `${yyyy}-${mm}-${dd}`;
    }
    function updatePreview(){
      const name = document.getElementById('name').value.trim() || 'Your Name';
      const domain = document.getElementById('domain').value;
      const issuer = document.getElementById('issuer').value.trim() || 'Surat AI Lab';
      const date = document.getElementById('date').value || '2026-01-02';
      const id = document.getElementById('id').value.trim() || 'SAL-2026-001';
      const level = document.getElementById('level').value;
      document.getElementById('certName').textContent = name;
      document.getElementById('certDomain').textContent = `${domain} — ${level} Level`;
      document.getElementById('certIssuer').textContent = issuer;
      document.getElementById('certDate').textContent = date;
      document.getElementById('certId').textContent = id;
      document.getElementById('certLevel').textContent = level;
      document.querySelector('.ribbon').textContent = domain.toUpperCase();
    }
    document.getElementById('previewBtn').addEventListener('click', updatePreview);
    document.getElementById('pdfBtn').addEventListener('click', async () => {
      updatePreview();
      const cert = document.getElementById('certificate');
      const canvas = await html2canvas(cert, { scale: 2, backgroundColor: null, useCORS: true });
      const imgData = canvas.toDataURL('image/png');
      const { jsPDF } = window.jspdf;
      const pdf = new jsPDF({ orientation: 'landscape', unit: 'pt', format: 'a4' });
      const pageWidth = pdf.internal.pageSize.getWidth();
      const pageHeight = pdf.internal.pageSize.getHeight();

      const margin = 24;
      const availableWidth = pageWidth - margin * 2;
      const availableHeight = pageHeight - margin * 2;

      const imgWidth = canvas.width;
      const imgHeight = canvas.height;
      const ratio = Math.min(availableWidth / imgWidth, availableHeight / imgHeight);

      const renderWidth = imgWidth * ratio;
      const renderHeight = imgHeight * ratio;
      const x = (pageWidth - renderWidth) / 2;
      const y = (pageHeight - renderHeight) / 2;

      pdf.addImage(imgData, 'PNG', x, y, renderWidth, renderHeight);

      const name = document.getElementById('certName').textContent || 'certificate';
      const domain = document.getElementById('domain').value || 'AI';
      pdf.save(`${name.replace(/\s+/g,'_')}_${domain.replace(/\s+/g,'_')}_certificate.pdf`);
    });
  </script>
